Get system information

/context/settings

Parameters

None

Example

Request:

pp://staging/context/system

Success:

{
  "result": {
    "setting": {
      "system": {
        "allowOffline": false,
        "traceInterval": 10
      },
      "price": {
        "roundDigit": 2,
        "roundStrategy": "round",
        "currency": "CYN"
      }
    },
    "version": "0.0.3"
  },
  "success": true,
  "error": {}
}

Fail:

{
  "success": false,
  "error": {
    "code": 202,
    "message": "Cannot get settings."
  }
}
